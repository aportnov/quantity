178:     def Optimizer.process(params)
179:       raise "Incoming parameter has to respond to [] method" unless params.respond_to?(:[])
180:       
181:       #initialize local values
182:       coefficient = params[:coefficient] || 1
183:       dividends = [] + (params[:dividends] || [])
184:       divisors =  [] + (params[:divisors] || [])
185: 
186:       #create arrays of unit hierarchy bases..  
187:       base_dividends = base(dividends)
188:       base_divisors = base(divisors)
189: 
190:       #try to find matching units and optimize them
191:       base_dividends.each_index { | i |
192:         if index = base_divisors.index(base_dividends[i])
193:           coefficient *= dividends[i].to_base(1.0)
194:           coefficient /= divisors[index].to_base(1.0)
195:             
196:           #set 'used' units to nil, so they won't be used again  
197:           dividends[i] = divisors[index] = base_divisors[index] = nil
198:         end
199:       }          
200:          
201:       return {
202:         :coefficient => coefficient,
203:         :dividends => dividends.compact,
204:         :divisors => divisors.compact
205:       }
