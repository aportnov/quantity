    def to_base(value)
    
      unless @to_base_coef 
        @to_base_coef = @coefficient
        @dividends.each {|unit| @to_base_coef *= unit.to_base(1.0) }
        @divisors.each {|unit| @to_base_coef /= unit.to_base(1.0) }
      end
    
      value * @to_base_coef
    end
