  def test_quantity_info_conversion
    
    qty = 14.m.to_quantity(@calc)
    assert_not_nil(qty)
    assert_equal(14, qty.value)
    assert_equal([:m], qty[:dividends])
    
    qty = qty * @calc.quantity(5, :lb)
    assert(qty)
    assert_equal(14 * 5, qty.value)
    assert_equal([:m, :lb], qty[:dividends])
  end
