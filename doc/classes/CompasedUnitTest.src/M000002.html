  def test_to_base_conversion

    unit = ComposedUnit.new({
      :dividends => [@units[:oz]]
    })

    assert_in_delta(2834.9523125, unit.to_base(100), 0.0001)

    unit = ComposedUnit.new({
      :dividends => [@units[:lb]],
      :divisors => [@units[:m]]
    })

    assert_in_delta(45.359237, unit.to_base(100), 0.0001)
  
  end
