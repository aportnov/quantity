  def test_numeric_mixin
    info = 12.mm
    assert(info.kind_of?(QuantityInfo))
    assert_equal(12, info.value)
    assert_equal(:mm, info.unit_sym)

    info = 12.566.mm
    assert(info.kind_of?(QuantityInfo))
    assert_in_delta(12.566, info.value, 0.0001)
    assert_equal(:mm, info.unit_sym)
  end
