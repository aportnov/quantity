  def test_mixed_operation
    
    assert_nothing_raised(RuntimeError) {
      
      info = 10.mm * 10
      assert(info.kind_of?(OperationInfo))
      
      info = 10.mm + 10
      assert(info.kind_of?(OperationInfo))

      info = 10 + 34.mm
      assert(info.kind_of?(OperationInfo))
    
    }

  end
