  def test_name_parsing
    name = "mm*lb/cm*kg"
  
    parts = name.split(/\//)
    
    assert_equal(2, parts.size)
    assert_equal("mm*lb", parts[0])
    assert_equal("cm*kg", parts[1])
    
    name = "mm*lb"
    
    parts = name.split(/\//)

    assert_not_nil parts[0]    
    assert_nil parts[1]
  end
