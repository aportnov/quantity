  def test_different_calculation_approaches
    qty = @calc.exp(:m) {25.cm + 15.8.mm + 10.m}
    
    result = @calc.exp(:m){ 35.cm + qty }
    assert_in_delta(10.6158, result.value, 0.001)
    
    assert_in_delta(1076.58, (15.cm + result)[:value], 0.001)
  end
