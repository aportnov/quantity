  def test_calculator_expression_method
    qty = @calc.exp {5.lb}
    assert(qty.kind_of?(Quantity))
    
    qty = @calc.exp {12.mm + 34.7.cm}
    assert(qty.kind_of?(Quantity))
    
    qty = @calc.quantity(45, :cm)
    assert_equal(qty, @calc.exp(:cm) {qty})
  end
